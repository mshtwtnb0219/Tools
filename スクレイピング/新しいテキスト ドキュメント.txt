		function putAuctionInfo(json_data) {
			var aId = json_data.auctionId;
			if($("#categoryPath_"+aId).text() == ''){
				$("#categoryPath_"+aId).hide().show(100).text(json_data.categoryPath);
			}
			if(json_data.isOffer == 'true'){
				var offerTxt = '';
					offerTxt += '<a href="http://auctions.yahoo.co.jp/jp/show/offer_manager?aID='+aId+'" class="u b" target="_blank" >';
					offerTxt += json_data.offerNum;  	
					offerTxt += '\n('+json_data.unansweredOfferNum+')';  	
					offerTxt += '</a>';
				$("#offer_"+aId).html(offerTxt);
			}
			$("#pageView_"+aId).hide().show(100).text(json_data.pageView);
			if ( json_data.watchListNum ) $("#watchListNum_"+aId).hide().show(100).text(json_data.watchListNum);
			$("#unansweredQAndANum_"+aId).addClass('u').hide().show(100).text(json_data.unansweredQAndANum);
			if(json_data.bidOrBuy != "0"){
				$("#bidOrBuy_"+aId).hide().show(100).text("即決価："+json_data.bidOrBuy+"円");
			}
			if ( json_data.bids ) $("#bids_"+aId).hide().show(100).text(json_data.bids);

			if(json_data.totalHighestBidders != '0'){
				$("#currentWinnerId_"+aId).hide().show(100).text(json_data.highestBidder);
			}										
		}

		function refreshSellingInfo(){
		    var receive_ing_cnt = 0;
		    
	    	$(".auctionId").each(function(){
				var aid = $(this).val();
		    	$.ajaxQueue({
					url: '/MySelling/MySellingItemAuthJson.php',
					data: {auctionId: aid},
					cache: false,
					async: true,
					success: function (data) {
						json_data = $.parseJSON(data);
						receive_ing_cnt++;
						if (json_data.auctionId) {
							putAuctionInfo(json_data);
							delete json_data.watchListNum;// ウオッチと入札数が更新されなくなるので、キャッシュに入れない
							delete json_data.bids;
							lscache.set('ainf_'+json_data.auctionId, json_data, localStorage_expire);
						} else {
							$("#pageView_"+aid).html("エラー<br>\n("+json_data.error_code+")");
						}
					}
		    	});
	    	});
		}
